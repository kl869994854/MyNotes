\subsection{子空间和商空间的对偶}

\begin{theorem}
    设$E$和$F$都是赋范空间，并有$u\in \mathcal{B}(E,F)$。则
    \begin{enumerate}
        \item $\ker u^{*}=u(E)^{\perp}$.
        \item $\ker u=u^{*}(F^{*})_{\perp}$.
        \item $(\ker u^{*})_{\perp}=\overline{u(E)}^{w}=\overline{u(E)}^{||\cdot||}$.
        \item $(\ker u)^{\perp}=\overline{u^{*}(F^{*})}^{w^{*}}$.
    \end{enumerate}
\end{theorem}
